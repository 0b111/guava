015package com.google.common.base;
016
017import com.google.common.annotations.GwtCompatible;
018import com.google.errorprone.annotations.CanIgnoreReturnValue;
019import org.checkerframework.checker.nullness.compatqual.NullableDecl;
020
021/**
022 * Determines a true or false value for a given input; a pre-Java-8 version of {@link
023 * java.util.function.Predicate java.util.function.Predicate}.
024 *
025 * <p>The {@link Predicates} class provides common predicates and related utilities.
026 *
027 * <p>See the Guava User Guide article on <a
028 * href="https://github.com/google/guava/wiki/FunctionalExplained">the use of {@code Predicate}</a>.
029 *
030 * <h3>For Java 8+ users</h3>
031 *
032 * <p>This interface is now a legacy type. Use {@code java.util.function.Predicate} (or the
033 * appropriate primitive specialization such as {@code IntPredicate}) instead whenever possible.
034 * Otherwise, at least reduce <i>explicit</i> dependencies on this type by using lambda expressions
035 * or method references instead of classes, leaving your code easier to migrate in the future.
036 *
037 * <p>To use a reference of this type (say, named {@code guavaPredicate}) in a context where {@code
038 * java.util.function.Predicate} is expected, use the method reference {@code
039 * guavaPredicate::apply}. For the other direction, use {@code javaUtilPredicate::test}. A future
040 * version of this interface will be made to <i>extend</i> {@code java.util.function.Predicate}, so
041 * that conversion will be necessary in only one direction. At that time, this interface will be
042 * officially discouraged.
043 *
044 * @author Kevin Bourrillion
045 * @since 2.0
046 */
047@GwtCompatible
048public interface Predicate<T> {
049  /**
050   * Returns the result of applying this predicate to {@code input} (Java 8 users, see notes in the
051   * class documentation above). This method is <i>generally expected</i>, but not absolutely
052   * required, to have the following properties:
053   *
054   * <ul>
055   *   <li>Its execution does not cause any observable side effects.
056   *   <li>The computation is <i>consistent with equals</i>; that is, {@link Objects#equal
057   *       Objects.equal}{@code (a, b)} implies that {@code predicate.apply(a) ==
058   *       predicate.apply(b))}.
059   * </ul>
060   *
061   * @throws NullPointerException if {@code input} is null and this predicate does not accept null
062   *     arguments
063   */
064  @CanIgnoreReturnValue
065  boolean apply(@NullableDecl T input);
066
067  /**
068   * Indicates whether another object is equal to this predicate.
069   *
070   * <p>Most implementations will have no reason to override the behavior of {@link Object#equals}.
071   * However, an implementation may also choose to return {@code true} whenever {@code object} is a
072   * {@link Predicate} that it considers <i>interchangeable</i> with this one. "Interchangeable"
073   * <i>typically</i> means that {@code this.apply(t) == that.apply(t)} for all {@code t} of type
074   * {@code T}). Note that a {@code false} result from this method does not imply that the
075   * predicates are known <i>not</i> to be interchangeable.
076   */
077  @Override
078  boolean equals(@NullableDecl Object object);
079}
